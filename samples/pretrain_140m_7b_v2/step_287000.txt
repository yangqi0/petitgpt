class Solution:
    def factorial(self, n):
        """
        :type n: int
        :rtype: int
        """
        if n < 0 or n > 1:
            raise ValueError("Number must be greater than zero")

        result = 1
        for i in range(1, n + 1):
            result *= i

        return result
