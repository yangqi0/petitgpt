def factorial_to_decimal(number):
    if len(number) == 0:
        return 0

    for i in range(0, len(number)):
        if number[i] != number[i + 1]:
            return number[i - 1] * (number[i] / 2) + number[i], number[i]]
    return number[len(number) // 2:]
