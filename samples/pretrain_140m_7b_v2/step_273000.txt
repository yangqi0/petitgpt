class Solution:
    def factorial(self, n):
        if not n:
            return 0

        result = [0] * (n + 1)
        for i in range(1, n + 1):
            result.append(i)
        
        for i in xrange(1, len(result)):
            result[i] += result[i - 1]
            
        return result

    def factorial_by_number(self, number):
