class Solution:
    def isNine(self, n):
        """
        :type n: int
        :rtype: bool
        """

        if not n:
            return True

        for i in range(0, len(n), 2):
            if n[i] == 1 and n[i-1] == 1:
                return False

            if n[n[i]] > 1:
                n = n[n[-1]] + 1

            if n < n[n+1]:
                return False

        return True
