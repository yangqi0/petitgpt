def factorial(n):
     return n*n


class Solution:
    def factorial(self, n):
        if not n:
            return 0

        n = n * (n + 1)

        for i in range(1, n+1):
            for j in range(i+1, n-1, -1):
                if n % j == 0:
                    break

                n -= 1

            if n > 0:
                return n / n

    def factorial_from_string(self, string):
        """
        :type string: str
        :rtype: int
        """
        return self.factorial(int(string))
