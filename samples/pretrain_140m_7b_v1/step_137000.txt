def factorial_count(number):
    if not isint(number):
        return 0

    # Get the factorial of the number
    factorial = 0
    for i in range(0, len(number)):
        if number[i] != number[i]:
            break

        if number[number[i]] > number[i + 1]:
            break
        else:
            factorial += 1

    return factorial
